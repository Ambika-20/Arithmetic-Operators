# Arithmetic-Operators
n1 = int(input("enter num:"))
n2 = int(input("enter num:"))
sum = n1+n2
sub = n1-n2
mul = n1*n2
div = n1/n2
floordiv = n1//n2
mod = n1%n2
exp = n1**n2
print("sum =",sum)
print("subtraction =",sub)
print("multiplication =",mul)
print("division =",div)
print("Floor Division =",floordiv)
print("modulus =",mod)
print("exponential =",exp)
